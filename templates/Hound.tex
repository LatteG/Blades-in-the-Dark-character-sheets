\StrCount{\SpecialAbilities}{ghost hunter}[\GhostHunterAmnt]
\ifnum\GhostHunterAmnt>0
    \def\GhostHunterY{-5.24cm}
    \tikzmath{\GhostHunterNum = min(1, \GhostHunterAmnt - 1);}
    
    \foreach \GhostHunterDot in {0,...,\GhostHunterNum}
        \fill[black] (\SpecialX + \SpecialXOffset * \GhostHunterDot, \GhostHunterY) circle (\SpecialR);
    
    \def\PetAbilityY{-6.02cm}
    
    % Pet abilities
    \IfSubStr{\SpecialAbilities}{ghost hunter:ghost-form}{%
                                 \UnderlineTikz{17.125cm}{1.325cm}{\PetAbilityY}}{}
    \IfSubStr{\SpecialAbilities}{ghost hunter:mind-link}{%
                                 \UnderlineTikz{18.625cm}{1.225cm}{\PetAbilityY}}{}
    \IfSubStr{\SpecialAbilities}{ghost hunter:arrow-swift}{%
                                 \UnderlineTikz{20.375cm}{1.400cm}{\PetAbilityY}}{}
    \tikzmath{\SpecialAbilitiesAmnt = \SpecialAbilitiesAmnt + \GhostHunterAmnt;}
\fi
